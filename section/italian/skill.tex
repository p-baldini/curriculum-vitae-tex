\section{Competenze trasversali}

\vspace{-\baselineskip}

\begin{multicols}{2}

\skill
    {Creatività e risoluzione di problemi}{-1}
    {
        Amo i giochi di logica e competitivi fin da bambino.
        Ho affinato la mia passione durante l'università.
    }

\skill
    {Organizzazione}{-1}
    {
        La mia formazione ingegneristica richiedeva un elevato livello di organizzazione di dati e attività.
    }

\skill
    {Crescita personale}{-1}
    {
        Cerco costantemente di imparare di più e di utilizzare le informazioni per produrre qualcosa di nuovo.
        Il mio dottorato è un chiaro esempio di questo atteggiamento.
    }

\skill
    {Lavoro di gruppo e ``leadership''}{-1}
    {
        Il mio impiego accademico richiedeva una collaborazione e una supervisione costanti.
        Un chiaro risultato di questo aspetto sono le varie pubblicazioni e tesi di cui sono stato (co-)autore e/o supervisore.
    }

\skill
    {Autonomia e iniziativa}{-1}
    {Lavorare con molti gruppi diversi a volte richiede di prendere l'iniziativa e di operare in autonomia.}

\end{multicols}
